Lets Encrypt Cluster Issuer with Cloudflare DNS Challenge

  • cloudflare API token needs Zone:Zone:Read, Zone:DNS:Edit on all zones(?)
apiVersion: cert-manager.io/v1alpha2
kind: ClusterIssuer
metadata:
  name: le-issuer
spec:
  acme:
    email: admin@example.com
    server: https://acme-v02.api.letsencrypt.org/directory
    privateKeySecretRef:
      name: le-issuer-account
    solvers:
      - dns01:
          cloudflare:
            email: admin@example.com
            apiTokenSecretRef:
              name: cloudflare
              key: token

certificate resource:

apiVersion: cert-manager.io/v1alpha2
kind: Certificate
metadata:
  name: api.seankhliao.com
spec:
  secretName: api-seankhliao-com-tls
  duration: 2160h
  renewBefore: 360h
  dnsNames:
    - api.seankhliao.com
    - "*.api.seankhliao.com"
  issuerRef:
    name: le-issuer
    kind: ClusterIssuer
